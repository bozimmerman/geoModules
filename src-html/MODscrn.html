;*******************************
;* ClrScrn
;*
;* Inputs:
;*	NONE
;*	
;* Outputs:
;*	.a	?
;*	.x	?
;*******************************

ClrScrn:	mvw	rightMargin,SCRNclr+7
	ldw	R0,#SCRNclr
	jmp	GraphicsString

SCRNclr:	.byte	NEWPATTERN,2
	.byte	MOVEPENTO
	.word	0
	.byte	0
	.byte	RECTANGLETO
	.word	319
	.byte	199
	.byte	0

;*******************************
;* OKCANBox
;*
;* Inputs:
;*	A2	Pointer to ok/cancel string
;*	
;* Outputs:
;*	.a	0
;*	.x	0
;*******************************

OKCANBox:	
	ldw	R0,#SCRNokc
	jmp	DoDlgBox
SCRNokc:	.byte	$80 | $01
	.byte	12
	.byte	$10,$1b
	.byte	A2
	.byte	$01,$01,$47	;OK icon
	.byte	$02,$0f,$47	;CANCEL icon
	.byte	0	;end of defintion


;*******************************
;* i_OKCANBox
;*
;* Inputs:
;*	.word	Pointer to ok/cancel string
;*	
;* Outputs:
;*	.a	0
;*	.x	0
;*******************************

i_OKCANBox:
	jsr	APIinit
	jsr	OKCANBox
	ldy	#2	; 1 words = 1 bytes
	jmp	APIreturn
