 sta dest+1
 lda #[(value)
 sta dest+0
.endm

.endm

 sta dest
.endm

.macro avw value,dest	;avw VALUE in-to DEST
 clc
 lda #[(value)
 adc dest+0
 sta dest+0
.if (value >= 0) && (value <= 255)
 bcc noInc
 inc dest+1
noInc:
.else
 lda #](value)
 adc dest+1
 sta dest+1
.endif
.endm

.endif
.endm
