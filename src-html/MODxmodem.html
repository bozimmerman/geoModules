;*               (c) 2017 Bo Zimmerman

;* Intro:
;*     A module for sending and receiving data using
;*     the X-Modem protocol.  Supports CHK and CRC16.
;*     Requires MODmodem or compatible library
;* 
;* Functions:
;* 	XMOrcvChk(	.word	Vector - Receive a single byte in .a
;*		.word	Vector - Received a good block
;*		.word	Vector - Received a bad block
;*		.word	Vector - Abort check (carry set = abort)
;*		) .x = 0 no error
;* 	XMOrcvCrc(	.word	Vector - Receive a single byte in .a
;*		.word	Vector - Received a good block
;*		.word	Vector - Received a bad block
;*		.word	Vector - Abort check (carry set = abort)
;*		) .x = 0 no error
;* 	XMOsend(	.word	Vector - Ensure 128 bytes available
;*		.word	Vector - Send a single byte in .a, .x=no more
;*		.word	Vector - Sent a bad block
;*		.word	Vector - Abort Check (carry set = abort)
;*		) .x = 0 no error
;*
;*     * = includes i_Function call, and returns
;*            status in the .x register
;*
;* Comments:
;*
;*********************************************

XMO_ACK	= $06
XMO_CAN	= $18
XMO_EOT	= $04
XMO_NAK	= $15
XMO_SOH	= $01
XMO_CNAK	= $43
XMO_PAD	= $1A
XMO_OTH	= $02

XMOSPut:	jmp ComSPut
XMOGet:	jmp ComGet
XMOnline:	jmp ComOnline
XMOPause:	jmp PauSec
XMOBufR:	jmp CLL2xxRb

;****************************************
;* XMOrcvChk		X-Modem Receive (checksum)
;* XMOrcvCrc		X-Modem Receive (CRC-16)
;* Inputs:
;*	A2	Vector - Receive a single byte in .a
;*	A3	Vector - Received a good block
;*	A4	Vector - Received a bad block
;*	A5	Vector - Abort check (carry set = abort)
;* Outputs:
;*	.x 	0 means A-OK, or abort/error
;*
;********************************
XMOrcvChk:	ldb	XMOXType,#0
	ldb	XMOBufSize,#$84
	jmp	XMOrcvSt
XMOrcvCrc:	ldb	XMOXType,#1
	ldb	XMOBufSize,#$85
XMOrcvSt:	mvw	A2,XMORcVec+1	; receive 1 byte in .a
	mvw	A3,XMORgVec+1	; receive good block
	mvw	A4,XMOErVec+1	; receive error block
	mvw	A5,XMOabVec+1	; abort check
	ldb	XMOUpDn,#1
	jsr	XMOClear
	jsr	XMOBufFlush
	inc	XMOBlock
	lda	XMOXType
	bne	10$
	lda	#XMO_NAK
	bne	20$
10$:	lda	#XMO_CNAK
20$:	jsr	XMOSPut
	jmp	XMOrcvLp

XMORcVec:	jmp	XMONada	; receive 1 byte
XMORgVec:	jmp	XMONada	; good block receive bytes
XMOErVec:	jmp	XMONada	; error block receive bytes
XMONada:	rts

XMOrcvLp:	lda	XMOErrors
	cmp	#$0e
	bcc	11$
10$:	ldb	XMOAbort,#$01
11$:	jsr	XMOChkAbort
	jsr	XMOChkTm
	lda	XMOAbort
	beq	15$
	jmp	XMOXAbort
14$:	jmp	XMOXRcv
15$:	cbi	XMOBlock,#$01
	bne	17$
	cbi	XMOTmOuts,#$02
	bne	17$
	beq	10$	; to cancel
17$:	lda	XMOClock
	beq	18$
	jmp	XMONoBort
18$:	lda	XMOBuf
	cmp	#XMO_CAN
	beq	10$
	cmp	#XMO_EOT
	beq	14$
	jsr	XMOgetKey
	lda	XMOKeySync
	beq	XMOrcvLp
	jsr	XMOReTm
	ldy	XMOBufBit
	lda	XMOKeyFree
	sta	XMOBuf,y
	inc	XMOBufBit
	iny
	cpy	XMOBufSize
	bcc	XMOrcvLp	; 
	jsr	XMOPause	; Check the Block
	lda	XMOBuf
	cmp	#XMO_EOT
	beq	14$
	cmp	#XMO_CAN
	beq	10$
	cmp	#XMO_OTH
	beq	20$
	cmp	#XMO_SOH
	beq	20$
19$:	jmp	XMOConRcv
20$:	cpb	XMOBuf+1, XMOBlock
	bne	19$
	eor	#$ff
	cmp	XMOBuf+2
	bne	19$
	jsr	XMOchkBlk
	lda	XMOXType
	bne	22$
	cpb	XMOChkSum, XMOBufEnd
	beq	25$
21$:	jmp	XMOConRcv
22$:	cpb	XMOBufEnd, XMOCrc
	bne	21$
	cpb	XMOBufCrc,XMOCrc+1
	bne	21$
25$:	ldb	XMOGood,#1
	jmp	XMOConRcv

XMOConRcv:	lda	XMOGood
	beq	20$
	ldb	XMOBufCt,#0
10$:	ldy	XMOBufCt
	lda	XMOBuf3,y
	jsr	XMORcVec
	inc	XMOBufCt
	cbi	XMOBufCt,#$80
	bcc	10$
	jsr	XMOBufFlush
	jsr	XMORgVec
	lda	#XMO_ACK
	jsr	XMOSPut
	inc	XMOBlock
	jmp	XMOrcvLp
20$:	lda	XMOAbort
	beq	XMONoBort
	jmp	XMOXAbort
XMONoBort:	jsr	XMOBufFlush
	jsr	XMOerVec
	lda	XMOXType
	bne	30$
25$:	lda	#XMO_NAK
	jsr	XMOSPut
	bra	40$
30$:	lda	XMOBlock
	cmp	#$02
	bcs	25$
	lda	#XMO_CNAK
	jsr	XMOSPut
40$:	inc	XMOErrors
	jmp	XMOrcvLp

XMOXRcv:	lda	#XMO_ACK
	jsr	XMOSPut
	jmp	XMOExit

XMOXAbort:	lda	#$01
	sta	XMOAbort
	lda	#XMO_CAN
	jsr	XMOSPut
	lda	#XMO_CAN
	jsr	XMOSPut
	lda	#XMO_CAN
	jsr	XMOSPut
	jmp	XMOExit

XMOExit:	ldx	XMOAbort
	jsr	XMOClear
	stx	XMOAbort
	jsr	XMOBufFlush
	lda	#0
	sta	XMOXType
	sta	XMOUpDn
	lda	XMOAbort
	tax
	rts


;****************************************
;* XMOsend		X-Modem Send
;* Inputs:
;*	A2	Vector - Ensure 128 bytes available
;*	A3	Vector - Send a single byte in .a, .x=no more
;*	A4	Vector - Sent a bad block
;*	A5	Vector - Abort Check (carry set = abort)
;* Outputs:
;*	.x 	0 means A-OK, or abort/error
;*
;********************************

XMOsend:	ldb	XMOUpDn,#0	; flag for upload
	jsr	XMOClear	; reset all the vars
	;jsr	XMOBufFlush	; this makes no sense!
	jsr	XMOgosCrc	; default CRC setup
	mvw	A2,XMOSsVec+1	; setup our vectors
	mvw	A3,XMOSnVec+1
	mvw	A4,XMOErVec+1
	mvw	A5,XMOabVec+1
	jmp	XMOsendStart	; start the upload

XMOSsVec:	jmp	XMONada	; prepare 128 bytes
XMOSnVec:	jmp	XMONada	; send 1 byte
XMOabVec:	jmp	XMONada

XMOgosChk:	ldb	XMOXType,#$00
	ldb	XMOBufSize,#$84
	rts
XMOgosCrc:	ldb	XMOXType,#$01
	ldb	XMOBufSize,#$85
	rts



XMObuild:	jsr	XMOSsVec
	ldy	#0
	sty	XMOIOStat
	sty	XMOBufCt
10$:	jsr	XMOSnVec
	ldy	XMOBufCt
	sta	XMOBuf3,y
	stx	XMOIOStat
	inc	XMOBufCt
	cpx	#0
	bne	20$
	cbi	XMOBufCt,#$80
	bcc	10$
	bra	30$
20$:	nop		; BZ: why? inc 	XMOBufCt
25$:	ldy	XMOBufCt
	lda	#XMO_PAD
	sta	XMOBuf3,y
	inc	XMOBufCt
	cbi	XMOBufCt,#$80
	bcc	25$
30$:	ldb	XMOBuf,#XMO_SOH
	lda	XMOBlock
	sta	XMOBuf+1
	eor	#$ff
	sta	XMOBuf+2
	jsr	XMOchkBlk
	lda	XMOChkSum
	sta	XMOBufEnd
	lda	XMOXType
	beq	XMOcontSend
	mvb	XMOCrc,XMOBufEnd
	mvb	XMOCrc+1,XMOBufCrc
XMOcontSend:	jsr	XMOminFlush
	ldy	#0
	sty	XMOBufBit
50$:	lda	XMOBuf,y
	jsr	XMOSPut
	ldy	XMOBufBit
	iny
	sty	XMOBufBit
	cpy	XMOBufSize
	bcc	50$
	jmp	XMOsendStart

XMOsendStart:	jsr	XMOReTm
10$:	jsr	XMOChkAbort
	jsr	XMOChkTm
	jsr	XMOGet
	bcs	19$
	cmp	#XMO_NAK
	beq	33$
	cmp	#XMO_CNAK
	beq	32$
	cmp	#XMO_CAN
	beq	30$
	cmp	#XMO_ACK
	beq	50$
19$:	cbi	XMOErrors,#$0e
	beq	30$
	lda	XMOAbort
	bne	30$
	lda	XMOClock
	bne	40$
29$:	beq	10$
30$:	jmp	XMOXAbort
32$:	jsr	XMOgosCrc
	bra	35$
33$:	jsr	XMOgosChk
35$:	lda	XMOBlock
	beq	50$
40$:	inc	XMOErrors
	jsr	XMOErVec
	jsr	XMOPause
	lda	XMOBlock
	beq	10$
	jsr	XMOBufR
	jmp	XMOcontSend
50$:	lda	XMOBlock
	bne	51$
	ldb	XMOErrors,#0
	jsr	XMOBufR
51$:	inc	XMOBlock
	lda	XMOIOStat
	beq	80$
	ldy	#0
70$:	tya
	pha
	lda	#XMO_EOT
	jsr	XMOSPut
	pla
	tay
	iny
	cpy	#$85
	bne	70$
	jmp	XMOExit
80$:	jsr	XMOPause
	jsr	XMOBufFlush
	jmp	XMObuild	

XMOClear:	lda	#0
	sta	XMOBlock
	sta	XMOGood
	sta	XMOErrors
	sta	XMOAbort
	sta	XMOBufBit
	sta	XMOChkSum
	sta	XMOCrc
	sta	XMOCrc+1
	sta	XMOTmOuts
	sta	XMOIOStat
	rts



XMOReTm:	jsr	Check128
	beq	10$
	phb	$01
	ldb	$01,#$35
	jsr	XMOClrTm
	plb	$01
	rts
10$:	phb	$ff00
	ldb	$ff00,#$7e
	jsr	XMOClrTm
	plb	$ff00
	rts

XMOChkTm:	ldb	XMOClock,#0
	jsr	Check128
	beq	20$
	phb	$01
	ldb	$01,#$35
	jsr	XMOGooTm
	plb	$01
10$:	cbi	XMOChkC,#3
	bcs	XMOBadTm
	rts
20$:	phb	$ff00
	ldb	$ff00,#$7e
	jsr	XMOGooTm
	plb	$ff00
	bra	10$
XMOBadTm:	inc	XMOTmOuts
	inc	XMOClock
	jmp	XMOReTm

XMOGooTm:	lda	$dc09
	beq	10$
	inc	XMOChkC
	lda	#0
	sta	$dc09
	sta	$dc08
10$:	lda	$dc08	; forces clock restart
	rts

XMOClrTm:	lda	#0
	sta	XMOChkC
	sta	$dc09
	sta	$dc08
	rts

XMOChkAbort:	clc
	jsr	XMOabVec
	bcs	10$
	jsr	XMOnline
	beq	10$
	lda	#0
	rts
10$:	ldb	XMOAbort,#1
	rts

XMOErrFlush:	inc	XMOErrors
XMOBufFlush:	ldy	#0
	tya
10$:	sta	XMOBuf,y
	iny
	cpy	#$85
	bcc	10$
XMOMinFlush:	;jsr	XMOBufR	; TODO: could this be a prob?
	lda	#0
	sta	XMOBufBit
	sta	XMOGood
	jsr	XMOReTm
	rts

XMOgetKey:	tya
	pha
	lda	#0
	sta	XMOKeySync
	jsr	XMOGet
	bcc	10$
	pla
	tay
	rts
10$:	sta	XMOKeyFree
	inc	XMOKeySync
	pla
	tay
	rts

XMOchkBlk:	lda	XMOXType
	bne	XMOchkCrc
	ldy	#0
	sta	XMOChkSum
10$:	lda	XMOBuf3,y
	add	XMOChkSum
	sta	XMOChkSum
	iny
	cpy	#$80
	bcc	10$
	rts

XMOchkCrc:	ldy	#0
	sty	XMOCrc
	sty	XMOCrc+1
10$:	lda	XMOBuf3,y
	eor	XMOCrc
	tax
	lda	XMOcrcMB,x
	eor	XMOCrc+1
	sta	XMOCrc
	lda	XMOcrcLB,x
	sta	XMOCrc+1
	iny
	cpy	#$80
	bne	10$
	rts
